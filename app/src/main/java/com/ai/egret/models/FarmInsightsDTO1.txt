package com.ai.egret.models


data class FarmInsightsResponse(
    val farm: FarmInsightsDTO,
    val current: CurrentDto,
    val trend: TrendDto,
    val advisory: AdvisoryDto,
    val history: List<HistoryItemDto>
)

data class FarmInsightsDTO(
    val id: Int,
    val name: String?,
    val area_m2: Double?,
    val created_at: String?,
    val crop: CropInfoDto?,
    val meta: Map<String, Any>?
)

data class CropInfoDto(
    val label: String?,
    val name: String?,
    val type: String?
)

data class CurrentDto(
    val farm_id: Int,
    val source: String?,
    val scene_id: String?,
    val scene_datetime: String?,
    val scene_age_human: String?,
    val health_score: Double?,
    val health_bucket: HealthBucketDto?,
    val ndvi_stats: NdviStatsDto?,
    val ndvi_level: NdviLevelDto?,
    val crop: CropInfoDto?
)

data class HealthBucketDto(
    val bucket: String?,
    val label: String?,
    val message: String?
)

data class NdviStatsDto(
    val min: Double?,
    val max: Double?,
    val mean: Double?,
    val median: Double?,
    val p10: Double?,
    val p90: Double?,
    val std: Double?
)

data class NdviLevelDto(
    val level: String?,
    val label: String?,
    val message: String?
)

data class AdvisoryDto(
    val severity: String?,
    val summary_messages: List<String>?,
    val recommended_actions: List<String>?
)

data class HistoryItemDto(
    val id: Int?,
    val scene_id: String?,
    val timestamp: String?,
    val health_score: Double?,
    val ndvi_mean: Double?,
    val ndvi_min: Double?,
    val ndvi_max: Double?
)

data class TrendPointDto(
    val timestamp: String?,
    val health_score: Double?,
    val ndvi_mean: Double?
)

data class TrendDto(
    val has_history: Boolean,
    val message: String?,
    val overall: String?,
    val points: List<TrendPointDto>?
)
